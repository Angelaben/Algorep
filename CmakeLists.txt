cmake_minimum_required (VERSION 2.6)
project (Algorep)
set (CMAKE_CXX_STANDARD 11)

add_executable(topologi src/project/Topologie/Topologie.cpp)

# Require MPI for this project:
find_package(MPI REQUIRED)
set(CMAKE_CXX_COMPILE_FLAGS ${CMAKE_CXX_COMPILE_FLAGS} ${MPI_COMPILE_FLAGS})
set(CMAKE_CXX_LINK_FLAGS ${CMAKE_CXX_LINK_FLAGS} ${MPI_LINK_FLAGS})
include_directories(MPI_INCLUDE_PATH)


target_link_libraries(topologi ${MPI_LIBRARIES})
# add the binary tree to the search path for include files
# so that we will find TutorialConfig.h
include_directories("src/project")
include_directories("src/project/Processus")

# add the executable
add_executable(Algorep src/project/firstExperiment.cpp)
add_executable(TestMemoryProcess Test/TestMemoryProcess.cpp src/project/Processus/MemoryProcess.cpp)
